#(extract) Run this when you add or change translatable text in Python or templates.
pybabel extract -F babel.cfg -o messages.pot .

# (update) Run this after extraction to update existing translations.
pybabel update -i messages.pot -d translations

# (compile) Compile Run this every time you edit a .po file.
pybabel compile -d translations

# (restart server) Restart the server is required after compile
